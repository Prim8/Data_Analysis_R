Assignments

m <- length(y)  # number of training examples
theta <- matrix(c(0, 0), nrow = 2)  # initialize coefficients (i.e. matrix of zeros)
num_iters <- 1500  # number of iterations
alpha <- 0.01  # learning rate
cost_history <- double(num_iters)  # vector of zeros of lenght num_iters
theta_history <- list(num_iters)  # lists thetas at each iteration step

Cost function (i.e. squared error function)

cost_function <- function(X, y, theta) {
    (1/(2 * m)) * t((X %*% theta) - y) %*% ((X %*% theta) - y)
}
cost_function(X, y, theta)

##       [,1]
## [1,] 32.07

Gradient descent update algorithm

for (i in 1:num_iters) {
    delta = t((1/m) * (t((X %*% theta) - y) %*% X))  # n x 1 matrix
    theta = theta - alpha * delta
    theta_history[[i]] <- theta
    cost_history[i] <- cost_function(X, y, theta)
}
print(theta)

##      [,1]
## x0 -3.630
## x1  1.166

solve(t(X) %*% X) %*% (t(X) %*% y)
